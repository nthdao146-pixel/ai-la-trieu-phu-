            // Hàm xáo trộn mảng

            function shuffle(array) {

                for (let i = array.length - 1; i > 0; i--) {

                    const j = Math.floor(Math.random() * (i + 1));

                    [array[i], array[j]] = [array[j], array[i]];

                }

            }

                const pollResults = {};

                allAnswers.forEach(ans => pollResults[ans] = 0);

                pollResults[currentQuestion.correctAnswer] = Math.floor(Math.random() * (70 - 40 + 1)) + 40; // Đáp án đúng luôn có % cao nhất

                remainingPercentage -= pollResults[currentQuestion.correctAnswer];

                const incorrectAnswers = allAnswers.filter(ans => ans !== currentQuestion.correctAnswer);

                if (incorrectAnswers.length > 0) {

                    const firstIncorrect = incorrectAnswers[0];

                    pollResults[firstIncorrect] = Math.floor(Math.random() * remainingPercentage);

                    remainingPercentage -= pollResults[firstIncorrect];

                }

                if (incorrectAnswers.length > 1) {

                    pollResults[incorrectAnswers[1]] = Math.floor(Math.random() * remainingPercentage);

                    remainingPercentage -= pollResults[incorrectAnswers[1]];

                }

                if (incorrectAnswers.length > 2) {

                    pollResults[incorrectAnswers[2]] = remainingPercentage;

                for (const [answer, percent] of Object.entries(pollResults)) {

                    popoverContent += `<p class="mb-2">${answer}: ${percent}%</p>`;

                }

                popoverContent += `</div>`;

            });

            lifelinePhoneBtn.addEventListener('click', () => {

                if (!lifelines['phone']) return;


                lifelines['phone'] = false;

                lifelinePhoneBtn.disabled = true;


                const currentQuestion = questionsShuffled[currentQuestionIndex];

                const friendAdvice = `嗯... 我认为答案是 ${currentQuestion.correctAnswer}。祝你好运！`;

                

                createPopover("亲友的建议:", `<p>${friendAdvice}</p>`);

            });


            // Tạo hộp thoại popover

            function createPopover(title, content) {

                popoverContainer.innerHTML = `

                    <div class="lifeline-popover">

                        <h3 class="text-2xl font-bold mb-4">${title}</h3>

                        <div>${content}</div>

                        <button id="close-popover" class="mt-6 bg-blue-600 hover:bg-blue-700 text-white font-semibold py-2 px-6 rounded-full">关闭</button>

                    </div>`;

                popoverContainer.classList.remove('hidden');


                document.getElementById('close-popover').addEventListener('click', () => {

                    popoverContainer.classList.add('hidden');

                });

            }


            // Bắt đầu trò chơi

            function startGame() {

                startScreen.classList.add('hidden');

                endScreen.classList.add('hidden');

                quizScreen.classList.remove('hidden');

                

                currentQuestionIndex = 0;

                currentPrize = 0;

                questionsShuffled = [...questions];

                shuffle(questionsShuffled);

                

                // Khôi phục quyền trợ giúp

                lifelines = { '5050': true, 'audience': true, 'phone': true };

                lifeline5050Btn.disabled = false;

                lifelineAudienceBtn.disabled = false;

                lifelinePhoneBtn.disabled = false;

                

                showQuestion();

            }


            // Kết thúc trò chơi và hiển thị kết quả

            function endGame(isWinner) {

                quizScreen.classList.add('hidden');

                endScreen.classList.remove('hidden');

                if (isWinner) {

                    finalPrizeDisplay.textContent = `恭喜！你赢得了 1,000,000 元！`;

                } else {

                    finalPrizeDisplay.textContent = `你带走的奖金是: ${prizeLevels[currentQuestionIndex -1] || 0} 元。`;

                }

            }


            // Gắn các sự kiện click

            startBtn.addEventListener('click', startGame);

            restartBtn.addEventListener('click', startGame);


            // Bắt đầu màn hình ban đầu

            startScreen.classList.remove('hidden');

        });
